Initializing the API


With the computed proof generated in prevous step we can initialize the Libris API. 
 <html>
  <head>
    <title>My Libris-SDK integration</title>    
    <script id="libris-jssdk" type="text/javascript" src="https://stage.bluelibris.com/static/js/libris-sdk.js"></script>
    <script  type="text/javascript">
      var initializeApi = function(){
        LibrisAPI.init({
          "application_id": "{{app.id}}",
          "nonce": "{{app.nonce}}",
          "proof": "{{app.proof}}",
          "debug": true,          
          "jsonp_timeout": 5000,
          "postback": "your_postback_url",          
          }, function(d) {
            console.log(d);
          }
        }
    </script>
    ...
  </head>
  <body onLoad="initializeAPI();">

  </body>
</html>
